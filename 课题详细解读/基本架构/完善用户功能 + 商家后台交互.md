## 路由配置

```dart

// 关注列表
GoRoute(
  path: '/following',
  name: 'following',
  builder: (context, state) => const FollowingPage(),
),

// 消息列表
GoRoute(
  path: '/messages',
  name: 'messages',
  builder: (context, state) => const MessageListPage(),
),

// 消息详情
GoRoute(
  path: '/message/:id',
  name: 'messageDetail',
  builder: (context, state) => MessageDetailPage(messageId: state.pathParameters['id']!),
),

// 编辑个人资料
GoRoute(
  path: '/edit-profile',
  name: 'editProfile',
  builder: (context, state) => const EditProfilePage(),
),

// 商品详情（商家后台）
GoRoute(
  path: '/admin/product/:id',
  name: 'adminProductDetail',
  builder: (context, state) => ProductDetailPage(productId: state.pathParameters['id']!),
),
```
