    └── SessionViewModel.kt # 会话列表逻辑

### 2.2设计思考与实践

- **分层原则**：遵循 MVI 架构，实现 “UI 与逻辑解耦、数据与业务解耦”。ViewModel 仅处理业务逻辑，不持有 UI 引用；Repository 统一封装数据操作，ViewModel 无需关心数据来源（本地 / 远程）。
- **单一职责**：每个模块仅负责一类功能（如 ChatViewModel 只处理消息相关逻辑，SessionViewModel 只处理会话管理），避免 “大而全” 的 ViewModel 导致维护困难。
- **扩展性考量**：通过接口定义 Repository（如 SessionRepository/RemoteSessionRepository），可快速替换数据实现。

## 4.核心功能技术实现分析

### 4.1状态管理实现

核心设计：不可变状态 + 单向更新

**代码示例（ChatViewModel 状态管理）**

```Kotlin
// 定义UI状态（不可变数据类）
data class ChatUiState(
    val messages: MutableList<ChatMessage> = mutableListOf(),
    val inputText: String = "",
    val isLoading: Boolean = false,
    val errorMessage: String? = null,
    val currentSessionId: String = ""
)

// 状态容器（私有可变，公有只读）
private val _uiState = MutableStateFlow(ChatUiState())
val uiState: StateFlow<ChatUiState> = _uiState.asStateFlow()

// 状态更新（通过update方法，确保线程安全）
_uiState.update { currentState ->
    currentState.copy(
        messages = updatedMessages,
        isLoading = false
    )
}
```

**设计思考**：

- 不可变状态：`ChatUiState`为数据类，所有字段通过`copy`方法更新，避免直接修改导致的状态混乱。
- 线程安全：`StateFlow`默认在主线程收集，ViewModel 中通过`viewModelScope`确保状态更新在协程中执行，避免 ANR。
- 状态隔离：ChatUiState 仅包含聊天页相关状态，SessionUiState 仅包含会话列表状态，避免状态污染。

### 4.2异步处理实现（协程 + Flow）

核心设计：协程调度 + 数据流回调

**代码示例（发送消息异步逻辑）**：

```JavaScript
viewModelScope.launch(Dispatchers.IO) {
    try {
        // 异步调用AI接口（IO线程）
        val aiMessage = chatRepository.sendMessage(text, sessionId)
        // 更新状态（主线程自动切换）
        _uiState.update { currentState ->
            currentState.copy(
                messages = updatedMessages,
                isLoading = false
            )
        }
    } catch (e: Exception) {
        // 异常处理
        _uiState.update { currentState ->
            currentState.copy(
                messages = updatedMessages,
                isLoading = false,
                errorMessage = e.message
            )
        }
    }
}
```

**设计思考**：

- 协程调度：`Dispatchers.IO`处理网络 / 数据库操作，避免阻塞主线程；ViewModelScope 自动跟随 ViewModel 生命周期，防止内存泄漏。
- Flow 冷订阅：`chatRepository.getMessages(sessionId)`返回 Flow，仅在 UI 收集时才执行查询，减少不必要的资源消耗。
- 异常处理：所有异步操作包裹 try-catch，异常信息通过 StateFlow 传递到 UI 层，统一展示错误提示。

### 4.3数据存储实现（Room 数据库）

核心设计：表关联 + 数据持久化

**代码示例（会话 - 消息关联设计）**：

```Kotlin
// 会话表
@Entity(tableName = "sessions")
data class Session(
    @PrimaryKey val id: String = UUID.randomUUID().toString(),
    val title: String,
    val userId: String,
    val createTime: Long,
    val lastMessageTime: Long
)

// 消息表（通过sessionId关联会话）
@Entity(tableName = "chat_messages")
data class ChatMessage(
    @PrimaryKey val id: String = UUID.randomUUID().toString(),
    val role: MessageRole,
    val content: String,
    val type: MessageType,
    val status: MessageStatus,
    @ColumnInfo(name = "sessionId") val sessionId: String, // 关联字段
    val timestamp: Long
)

// DAO查询（按sessionId过滤消息）
@Query("SELECT * FROM chat_messages WHERE sessionId = :sessionId ORDER BY timestamp ASC")
fun getMessagesBySessionId(sessionId: String): Flow<List<ChatMessage>>
```

**设计思考**：

- 表关联设计：通过`sessionId`实现 “会话 - 消息” 的一对多关联，符合关系型数据库设计范式，避免数据冗余。
- 数据流查询：DAO 方法返回 Flow，数据库数据变更时自动回调，实现消息列表的实时更新。
- 数据迁移：新增字段（如 timestamp）时，通过 Room Migration 实现平滑升级，避免数据丢失。

### 4.4上下文对话功能实现

核心设计：上下文拼接 + 会话隔离

**代码示例（上下文拼接逻辑）**

```Kotlin
private suspend fun getHistoryMessagesForContext(sessionId: String): List<ChatMessage> {
    return chatDao.getMessagesBySessionId(sessionId)
        .first()
        .filter { it.status == MessageStatus.SUCCESS }
        .sortedBy { it.timestamp }
        .takeLast(MAX_CONTEXT_MESSAGES) // 限制上下文条数
}

// 发送消息时拼接上下文
val contextMessages = getHistoryMessagesForContext(sessionId) + userMessage
val apiMessages = contextMessages.map { it.toApiMessage() }
val request = ChatCompletionRequest(messages = apiMessages)
```

**设计思考**：

- 会话隔离：每个会话的上下文独立拼接，通过`sessionId`过滤，避免不同会话的消息混淆。
- 性能优化：限制上下文最大条数（如 20 条），避免 AI 接口 token 超标；过滤失败 / 加载中的消息，确保上下文有效性。
- 体验优化：上下文拼接逻辑封装在 Repository 层，ViewModel 无需关心细节，符合 “关注点分离” 原则。

## 5.实践总结与优化方向

### 5.1实践总结

- 分层架构的价值：MVI分层使得功能迭代（如新增视频生成、标题自定义）时，仅需修改对应层代码，无需全量重构。
- 异步处理的关键：协程 + Flow 是 Android 异步编程的最优解，尤其是在 AI 应用中，大量的网络 / 数据库操作需要非阻塞处理。
- 数据存储的核心：关系型数据库（Room）更适合 “会话 - 消息” 的结构化数据，相较于 NoSQL（如 MMKV），查询灵活性更高。

### 5.2优化方向

- 性能优化：引入分页加载，避免大量消息导致的 LazyColumn 卡顿；
- 功能扩展：支持上下文摘要（**长对话时利用AI生成历史内容摘要**，减少 token 消耗，增加记忆存储量）、多用户切换、消息收藏，**中间引入结构化prompt层优化生成内容**。
- 架构升级：引入 Hilt 实现依赖注入，减少手动创建 ViewModel/Repository 的冗余代码；接入 Jetpack Compose Navigation 2.0，优化深层链接。

优化后

```Java
class RemoteChatRepository @Inject constructor(
    private val apiService: ApiService,
    private val chatDao: ChatDao
) : ChatRepository {
    // 最大上下文消息条数
    private val MAX_CONTEXT_MESSAGES = 20

    override suspend fun sendMessage(text: String, sessionId: String): ChatMessage {
        // 1. 存储用户消息到本地数据库
        val userMessage = ChatMessage(
            role = MessageRole.USER,
            content = text,
            type = MessageType.TEXT,
            status = MessageStatus.SUCCESS,
            sessionId = sessionId,
            userId = "default_user",
            timestamp = System.currentTimeMillis()
        )
        chatDao.insertMessage(userMessage)

        // 2. 获取并裁剪当前会话的历史上下文
        val contextMessages = getHistoryMessagesForContext(sessionId) + userMessage

        // 3. 转换为API要求的消息格式
        val apiMessages = contextMessages.map { it.toApiMessage() }

        // 4. 调用AI接口（IO线程）
        val request = ChatCompletionRequest(
            model = "gpt-3.5-turbo",
            messages = apiMessages,
            temperature = 0.7f
        )
        val response = apiService.createChatCompletion(request)
        if (!response.isSuccessful || response.body() == null || response.body()!!.choices.isEmpty()) {
            throw Exception("AI回复获取失败")
        }

        // 5. 存储AI回复到本地数据库
        val aiContent = response.body()!!.choices[0].message.content
        val aiMessage = ChatMessage(
            role = MessageRole.AI,
            content = aiContent,
            type = MessageType.TEXT,
            status = MessageStatus.SUCCESS,
            sessionId = sessionId,
            userId = "default_user",
            timestamp = System.currentTimeMillis()
        )
        chatDao.insertMessage(aiMessage)

        return aiMessage
    }

    // 私有工具方法：获取并裁剪上下文消息
    private suspend fun getHistoryMessagesForContext(sessionId: String): List<ChatMessage> {
        return withContext(Dispatchers.IO) {
            chatDao.getMessagesBySessionId(sessionId)
                .first() // Flow转一次性列表
                .filter { it.status == MessageStatus.SUCCESS } // 仅保留成功消息
                .sortedBy { it.timestamp } // 按时间升序
                .takeLast(MAX_CONTEXT_MESSAGES) // 限制最大条数
        }
    }

    // 扩展：ChatMessage转ApiMessage
    private fun ChatMessage.toApiMessage(): ApiMessage {
        return ApiMessage(
            role = when (this.role) {
                MessageRole.USER -> "user"
                MessageRole.AI -> "assistant"
            },
            content = this.content
        )
    }
}
```

通过`sessionId`实现会话隔离，裁剪无效消息和条数，避免上下文错乱和 token 超标；上下文拼接基于本地数据库的持久化消息，重启 App 后仍能保留历史对话记忆；完全复用现有Repository、DAO 方法，与消息发送流程完美衔接，集成后即可实现上下文对话体验。

// 实体类（Room注解，补充关联字段）
@Entity(tableName = "sessions")
data class Session(
    @PrimaryKey val id: String = UUID.randomUUID().toString(),
    var title: String = "新会话",
    val userId: String = "default_user",
    val createTime: Long = System.currentTimeMillis(),
    var lastMessageTime: Long = System.currentTimeMillis()
)

// SessionDao
@Dao
interface SessionDao {
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertSession(session: Session)

    @Query("SELECT * FROM sessions WHERE userId = :userId ORDER BY lastMessageTime DESC")
    fun getUserSessions(userId: String): Flow<List<Session>>

    @Delete
    suspend fun deleteSession(session: Session)

    @Query("SELECT * FROM sessions WHERE id = :sessionId")
    suspend fun getSessionById(sessionId: String): Session?
}

// Repository接口
interface SessionRepository {
    suspend fun createNewSession(userId: String): Session
    fun getUserSessions(userId: String): Flow<List<Session>>
    suspend fun deleteSession(sessionId: String)
    suspend fun getSessionById(sessionId: String): Session?
}

// Repository实现类
class RemoteSessionRepository @Inject constructor(
    private val sessionDao: SessionDao,
    private val chatDao: ChatDao // 用于级联删除消息
) : SessionRepository {
    override suspend fun createNewSession(userId: String): Session {
        val newSession = Session(userId = userId)
        sessionDao.insertSession(newSession)
        return newSession
    }

    override fun getUserSessions(userId: String): Flow<List<Session>> {
        return sessionDao.getUserSessions(userId)
    }

    override suspend fun deleteSession(sessionId: String) {
        // 级联删除关联消息
        chatDao.clearMessagesBySessionId(sessionId)
        sessionDao.getSessionById(sessionId)?.let { sessionDao.deleteSession(it) }
    }

    override suspend fun getSessionById(sessionId: String): Session? {
        return sessionDao.getSessionById(sessionId)
    }
}

// ViewModel（通过Repository获取数据）
class SessionViewModel @Inject constructor(
    private val sessionRepository: SessionRepository
) : ViewModel() {
    private val _uiState = MutableStateFlow(SessionUiState())
    val uiState: StateFlow<SessionUiState> = _uiState.asStateFlow()

    data class SessionUiState(
        val sessions: List<Session> = emptyList(),
        val selectedSessionId: String = "",
        val isLoading: Boolean = false,
        val errorMessage: String? = null
    )

    // 加载用户会话
    fun loadUserSessions(userId: String) {
        viewModelScope.launch(Dispatchers.IO) {
            sessionRepository.getUserSessions(userId).collectLatest { sessions ->
                _uiState.update { it.copy(sessions = sessions) }
            }
        }
    }

    // 创建会话
    fun createNewSession(userId: String) {
        viewModelScope.launch(Dispatchers.IO) {
            sessionRepository.createNewSession(userId)
        }
    }

    // 切换会话
    fun switchSession(sessionId: String) {
        _uiState.update { it.copy(selectedSessionId = sessionId) }
    }

    // 删除会话
    fun deleteSession(sessionId: String) {
        viewModelScope.launch(Dispatchers.IO) {
            sessionRepository.deleteSession(sessionId)
        }
    }
}

## 4.思考

### 4.1写好 Prompt 的 3 个核心要素

- 业务上下文清晰：必须明确 “当前开发阶段”“已实现功能”“核心业务目标”，避免 AI 生成脱离项目的通用代码；
- 技术约束具体：明确技术栈、架构模式、已有类 / 方法的复用要求，减少代码冲突；
- 输出标准明确：指定代码范围、命名规范、注释要求，甚至可以给出 “验收标准”，让 AI 输出更贴合实际需求。

### 4.2迭代优化的关键技巧

- 问题精准化：反馈时不笼统说 “不符合需求”，而是具体指出 “缺失 XX 字段”“未集成 XX 功能”；
- 增量补充：每次优化仅针对 1-2 个核心问题，逐步完善逻辑，避免一次性添加过多约束导致 AI 理解混乱；
- 上下文复用：将前序生成的正确代码片段融入后续 Prompt，让 AI 基于现有代码风格和逻辑迭代，保证项目代码的一致性。

总结：Vibe Coding 的本质不是 “让 AI 代替开发”，而是通过 “精准指令 + 迭代优化” 让 AI 成为 “高效协作伙伴”。
